function selectionSort(array) {
  // Only change code below this line

  let smallArr = []
  for(let i = array.length - 1 ; i >= 0; i--){
    let smallest = Math.min(...array)
    smallArr.push(smallest)
    array.splice(smallest,1)
  }




  return smallArr;
}
